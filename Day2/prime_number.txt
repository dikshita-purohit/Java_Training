//prime number

import java.util.*;
public class TestPrj2 {
	public static void main(String args[]) {
		Scanner sc=new Scanner(System.in);
		System.out.print("Enter a number: ");
		int j=sc.nextInt();
		int c=0;
		for(int i=2;i<j;i++){
			if(j%i==0) {
				c++;
			}
		}
		if(c==0)
			System.out.print("Prime");
		else
			System.out.print("Not prime");
	}
}
