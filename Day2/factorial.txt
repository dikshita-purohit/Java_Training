import java.util.*;
public class TestPrj2 {
	public static void main(String args[]) {
		Scanner sc=new Scanner(System.in);
		System.out.print("Enter n value: ");
		int s = sc.nextInt();
		
		if(s == 0) {
			System.out.print("1");
		}else if(s<0) {
			System.out.print("-1");;
		}else {
				int fact = 1;
				for(int i=1;i<=s;i++) {
					fact *= i;
				}
				System.out.print("Factorial is : "+fact);
			
		}
	}
}
